# LDEX core Logos module
cmake_minimum_required(VERSION 3.14)
project(LdexCorePlugin LANGUAGES CXX)

# LogosModule.cmake is provided by logos-module-builder via env in nix builds
if(DEFINED ENV{LOGOS_MODULE_BUILDER_ROOT})
    include($ENV{LOGOS_MODULE_BUILDER_ROOT}/cmake/LogosModule.cmake)
elseif(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/cmake/LogosModule.cmake")
    include(cmake/LogosModule.cmake)
else()
    message(FATAL_ERROR "LogosModule.cmake not found")
endif()

logos_module(
    NAME ldex_core
    SOURCES
        src/ldex_core_interface.h
        src/ldex_core_plugin.h
        src/ldex_core_plugin.cpp
    EXTERNAL_LIBS
        wallet_ffi
        ldex_amm_ffi
)
